019package org.jdrupes.taglets.plantUml;
020
021import java.util.List;
022import java.util.Set;
023
024import javax.lang.model.element.Element;
025
026import com.sun.source.doctree.DocTree;
027
028import jdk.javadoc.doclet.Taglet;
029
030/**
031 * A dummy taglet that provides no output.
032 */
033public class EndUml implements Taglet {
034
035    @Override
036    public String getName() {
037        return "enduml";
038    }
039
040    @Override
041    public Set<Location> getAllowedLocations() {
042        return Set.of(Taglet.Location.OVERVIEW, Taglet.Location.PACKAGE,
043            Taglet.Location.TYPE);
044    }
045
046    @Override
047    public boolean isInlineTag() {
048        return false;
049    }
050
051    @Override
052    public String toString(List<? extends DocTree> tags, Element element) {
053        return "";
054    }
055
056}
